# Usa la imagen oficial de .NET 8 como base
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS base
WORKDIR /app
EXPOSE 8080

# Build stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src
COPY ["SoundTrack.Api/SoundTrack.Api.csproj", "SoundTrack.Api/"]
RUN dotnet restore "SoundTrack.Api/SoundTrack.Api.csproj"
COPY . .
WORKDIR "/src/SoundTrack.Api"
RUN dotnet build "SoundTrack.Api.csproj" -c Release -o /app/build

FROM build AS publish
RUN dotnet publish "SoundTrack.Api.csproj" -c Release -o /app/publish

FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "SoundTrack.Api.dll"]
